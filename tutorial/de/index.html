API-Design ähnlich wie jQuery, aber intuitiver

      
// jQuery Code
$("#target").html("some html code"); // HTML setzen
$("#target").text("set text"); // Text setzen
var ele_text = $("#target").text(); // Text abrufen
var parents = $("#target").parents(); // Array aller übergeordneten Ebenen abrufen
var child = $("#target").children()[0]; // Erstes untergeordnetes Element abrufen
      
    
      
// ofa.js Code
$("#target").html = "some html code";  // HTML setzen
$("#target").text = "set text"; // Text setzen
var ele_text = $("#target").text; // Text abrufen
var parents = $("#target").parents; // Array aller übergeordneten Ebenen abrufen
var child = $("#target")[0]; // Erstes untergeordnetes Element abrufen
